# START — build + run all 9 containers (Postgres, Redis, migrate, 3 services, 2 gateways, Nginx)
docker compose up --build -d

# STOP — stop & remove containers (keeps the Postgres data volume)
docker compose down

# STOP + wipe the database volume too (fresh start next time)
docker compose down -v

# START infra — just the two datastores
docker compose up -d postgres redis

# STOP infra
docker compose stop postgres redis        # pause (keeps data)
# or: docker compose down                  # remove containers
